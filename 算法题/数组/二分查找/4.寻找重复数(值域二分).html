                        # 题目

                        寻找重复数open in new window

                        # 思路

                        题解open in new window 二分查找有两种：

                        1. 索引二分：一般用于有序数组中找元素，因为索引的大小可以反映值的大小，因此对索引二分即可
                        2. 值域二分：重复数落在 [1, n] ，可以对 [1, n] 这个值域二分查找

                        # 实现

                        var findDuplicate = function (nums) {
  let left = 1,
    right = nums.length - 1;
  while (left < right) {
    const mid = (left + right) >>> 1;
    let count = 0;
    for (let i = 0; i < nums.length; i++) {
      // 统计比mid小的数量
      if (nums[i] <= mid) {
        count++;
      }
    }
    // 说明要找的值在mid的左侧
    if (count > mid) {
      right = mid;
    }
    // 在mid右侧
    else {
      left = mid + 1;
    }
  }
  return left;
};
