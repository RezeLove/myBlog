                                    # 思路

                                    本质上与 11 题类似，是区间重叠的问题，该题要求移除区间的最小数量，实际上只要找到不重叠区间的数量，再用总数量减去该数量即可

                                    # 实现

                                    var eraseOverlapIntervals = function (intervals) {
  // 先按照右边界排序
  intervals.sort((a, b) => {
    return a[1] - b[1];
  });
  let count = 1,
    right = intervals[0][1];
  for (let i = 1; i < intervals.length; i++) {
    let cur = intervals[i];
    // 不重叠的情况
    if (cur[0] >= right) {
      count++;
      right = cur[1];
    }
  }
  // 总区间数量 - 不重叠区间数量
  return intervals.length - count;
};
