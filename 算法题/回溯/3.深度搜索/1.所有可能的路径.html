        var allPathsSourceTarget = function (graph) {
  const target = graph.length - 1;
  let res = [],
    path = [];
  // 当前的出发节点
  const dfs = (startIndex) => {
    if (startIndex === target) {
      res.push([...path]);
      return;
    }
    // 遍历当前出发节点能到达的节点
    for (let i = 0; i < graph[startIndex].length; i++) {
      path.push(graph[startIndex][i]);
      // 将这个节点作为下一个出发节点
      dfs(graph[startIndex][i]);
      path.pop();
    }
  };
  path.push(0);
  dfs(0);
  return res;
};
