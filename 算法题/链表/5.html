                # 题目连接

                两两交换链表元素open in new window

                # 思路

                两两交换链表除了改变相邻结点的 next 关系外，还需要改变前一个结点的 next，因此就是 3 个一组进行处理，可以采用滑动窗口方法

                # 实现

                滑动窗口

                var swapPairs = function (head) {
  let ue = new ListNode(0, head);
  let first, second, cur, temp;
  cur = ue;
  while (cur.next && cur.next.next) {
    first = cur.next;
    second = first.next;
    cur.next = second;
    temp = second.next;
    second.next = first;
    first.next = temp;
    // 完成交换

    // 移动到下一个窗口
    cur = first;
  }
  return ue.next;
};

                递归实现滑动窗口

                const moveExchange = (head) => {
  let first, second, temp;
  if (head.next && head.next.next) {
    first = head.next;
    second = head.next.next;
    head.next = second;
    temp = second.next;
    second.next = first;
    first.next = temp;
    moveExchange(first);
  }
  return;
};
var swapPairs = function (head) {
  let ue = new ListNode(0, head);
  moveExchange(ue);
  return ue.next;
};
