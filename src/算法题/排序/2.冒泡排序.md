```js
function bubbleSort(arr) {
  const len = arr.length;

  for (let i = 0; i < len - 1; i++) {
    // 标记是否发生了交换
    let swapped = false;

    for (let j = 0; j < len - 1 - i; j++) {
      // 如果前一项比后一项大，则交换它们
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        swapped = true;
      }
    }

    // 如果本轮遍历中没有发生交换，说明数组已经有序，直接退出循环
    if (!swapped) {
      break;
    }
  }

  return arr;
}
```
