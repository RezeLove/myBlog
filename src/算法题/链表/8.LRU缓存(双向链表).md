## 题目链接

[LRU 缓存](https://leetcode.cn/problems/lru-cache/)

## 思路

数据结构的选择：

1. map+array:map 存储 key 和 value，array 维护操作顺序。 缺点：array 的插入/移动/删除的复杂度都为 O(n)

2. map+双向链表：map 存储 key 和 key 对应的 node；双向链表存储 key 和 val。 优点：双向链表的移动和删除都是指针的变化，复杂度为 O(1)

## 实现

1. map + array

```js
var LRUCache = function (capacity) {
  this.cache = new Map();
  this.queue = [];
  this.len = capacity;
};

// 每次操作后，就将被操作的key重置到队尾
LRUCache.prototype.refresh = function (key) {
  let index = this.queue.indexOf(key);
  this.queue.splice(index, 1);
  this.queue.push(key);
};

LRUCache.prototype.get = function (key) {
  if (this.cache.has(key)) {
    this.refresh(key);
    return this.cache.get(key);
  } else {
    return -1;
  }
};

LRUCache.prototype.put = function (key, value) {
  if (this.cache.get(key)) {
    this.cache.set(key, value);
    // 被操作key重置到队尾
    this.refresh(key);
  }
  //   新插入的元素的key直接放到队尾
  else if (this.queue.length < this.len) {
    this.cache.set(key, value);
    this.queue.push(key);
  } else {
    // 队头的元素是最先被删除的
    let outkey = this.queue.shift();
    this.cache.delete(outkey);
    this.cache.set(key, value);
    this.queue.push(key);
  }
};
```

2. 双向链表

```js
var Node = function (key = 0, val = 0) {
  this.val = val;
  this.key = key;
  //   有prev和next
  this.prev = null;
  this.next = null;
};

class LRUCache {
  constructor(capacity) {
    this.capacity = capacity;
    this.dummy = new Node(); // 哨兵节点
    // dummy的prev和next都先指向自身
    this.dummy.prev = this.dummy;
    this.dummy.next = this.dummy;
    this.map = new Map();
  }
  // 删除某个node
  remove(x) {
    x.prev.next = x.next;
    x.next.prev = x.prev;
  }
  // 将某个node推到头部
  pushFront(x) {
    x.next = this.dummy.next;
    x.prev = this.dummy;
    this.dummy.next = x;
    x.next.prev = x;
  }
  get(key) {
    if (this.map.has(key)) {
      let node = this.map.get(key);
      //   先删除再推到头部
      this.remove(node);
      this.pushFront(node);
      return node.val;
    } else {
      return -1;
    }
  }
  put(key, value) {
    if (this.map.has(key)) {
      let node = this.map.get(key);
      node.val = value;
      this.remove(node);
      this.pushFront(node);
      return;
    }
    let node = new Node(key, value);
    this.map.set(key, node);
    this.pushFront(node);
    if (this.map.size > this.capacity) {
      // 超过容量，删除最后一个node
      let lastnode = this.dummy.prev;
      let lastkey = lastnode.key;
      this.remove(lastnode);
      this.map.delete(lastkey);
    }
  }
}
```
